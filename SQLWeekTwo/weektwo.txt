SELECT DISTINCT	COUNT(E.emp_no) 'employee count'	, T.title	FROM employees E INNER JOIN titles T	ON T.emp_no = E.emp_no 	WHERE E.birth_date > '1965-01-01'		GROUP BY 	 T.title//////////////////////SELECT DISTINCT	AVG(S.salary) 'average salary'	, T.title	FROM titles TINNER JOIN salaries S		ON T.emp_no = S.emp_no			GROUP BY 	 T.title/////////////////////////////////SELECT DISTINCT	SUM(S.salary)	FROM departments DINNER JOIN dept_manager DM	ON DM.dept_no = D.dept_no INNER JOIN salaries S	ON S.emp_no = DM.emp_no WHERE D.dept_no = 'd001' /*Marketing*/AND YEAR(S.from_date) BETWEEN '1990' AND '1992'
